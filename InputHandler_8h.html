 85/**
 86 * @brief User defined function parameters
 87 */
 88class UserCallbackFunctionParameters
 89{
 90public:
 91    uint16_t num_args;                                                 /** number of function arguments */
 92    const uint8_t *_arg_type;                                          /** function argument type array pointer */
 93    const char *command;                                               /** command to match */
 94    uint16_t command_length;                                           /** length of command */
 95    void (*function)(UserInput *);                                     /** pointer to function */
 96    UserCallbackFunctionParameters *next_callback_function_parameters; /** UserCallBackFunctionParameters iterator/pointer */
 97    /**
 98     * @brief UserCallbackFunctionParameters Constructor
 99     *
100     * Creates a new instance of this class.  Before using, construct a UserInput object.
101     * @param user_defined_command_to_match The command which when entered will call a function
102     * @param user_defined_function_to_call The function called when the command is matched
103     * @param args args is a variadic parameter pack of USER_INPUT_TYPE
104     */
105    template <typename... Arguments>
106    UserCallbackFunctionParameters(const char *user_defined_command_to_match,
107                                   void (*user_defined_function_to_call)(UserInput *),
108                                   const Arguments &...args)
109        : command(user_defined_command_to_match),
110          function(user_defined_function_to_call),
111          command_length(strlen_P(command)),
112          next_callback_function_parameters(NULL),
113          num_args(sizeof...(Arguments))
114    {
115        static const uint8_t _arg[] = {static_cast<uint8_t>(args)...}; /** args is expanded into the array _arg */
116
117        _arg_type = _arg; /** point to the array in memory */
118    }
119};

 42/*
 43   test all available input types
 44*/
 45void uc_test_input_types(UserInput *inputProcess)
 46{
 47  char *str_ptr = inputProcess->NextArgument();                                     //  init str_ptr and point it at the next argument input by the user
 48  char *strtoul_ptr = 0;                                                            //  this is for strtoul
 49  uint32_t strtoul_result = strtoul(str_ptr, &strtoul_ptr, 10);                     // get the result in base10
 50  uint8_t eight_bit = (strtoul_result <= UINT8_MAX) ? (uint8_t)strtoul_result : 0U; // if the result is less than UINT8_MAX then set eight_bit, else eight_bit = 0
 51
 52  str_ptr = inputProcess->NextArgument();
 53  strtoul_ptr = 0;
 54  strtoul_result = strtoul(str_ptr, &strtoul_ptr, 10);
 55  uint16_t sixteen_bit = (strtoul_result <= UINT16_MAX) ? (uint16_t)strtoul_result : 0U;
 56
 57  str_ptr = inputProcess->NextArgument();
 58  strtoul_ptr = 0;
 59  uint32_t thirtytwo_bit = strtoul(str_ptr, &strtoul_ptr, 10);
 60
 61  str_ptr = inputProcess->NextArgument();
 62  int sixteen_bit_int = atoi(str_ptr);
 63
 64  str_ptr = inputProcess->NextArgument();
 65  float thirtytwo_bit_float = (float)atof(str_ptr);
 66
 67  str_ptr = inputProcess->NextArgument();
 68  char _char = *str_ptr;
 69
 70  str_ptr = inputProcess->NextArgument();
 71  char c_string[64] = {'\0'};
 72  snprintf_P(c_string, 64, PSTR("%s"), str_ptr);
 73
 74  char float_buffer[32] = {'\0'}; //  dtostrf buffer
 75  char out[128] = {'\0'};         //  function output buffer
 76  uint16_t string_pos = 0;        // function output buffer index
 77
 78  /*
 79       format out[] with all of the arguments received
 80  */
 81  string_pos += snprintf_P(out + string_pos, 128,
 82                           PSTR("Test user input types:\n"
 83                                "uint8_t %u\nuint16_t %u\nuint32_t %lu\nint %d\nfloat %s\nchar %c\nc-string %s\n"),
 84                           eight_bit,
 85                           sixteen_bit,
 86                           thirtytwo_bit,
 87                           sixteen_bit_int,
 88                           dtostrf(thirtytwo_bit_float, 2, 3, float_buffer),
 89                           _char,
 90                           c_string);
 91  Serial.print(out);
 92}

101/*
102   UserInput UserCallbackFunctionParameters
103   These objects are what you use to specify the command string, function to launch, and types of input if any
104
105   The following are the available input types
106   USER_INPUT_TYPE_UINT8_T == an eight bit unsigned integer
107   USER_INPUT_TYPE_UINT16_T == a sixteen bit unsigned integer
108   USER_INPUT_TYPE_UINT32_T == a thirtytwo bit unsigned integer
109   USER_INPUT_TYPE_INT16_T == a sixteen bit signed integer
110   USER_INPUT_TYPE_FLOAT == a thirtytwo bit signed floating point number
111   USER_INPUT_TYPE_CHAR == a character value
112   USER_INPUT_TYPE_C_STRING == a string of character values, absent of '\0' and enclosed with single quotation marks "c-string"
113                               depending on the method used, if it is ReadCommand then very long c-strings can be sent and read
114                               using GetCommandFromStream input c-string length is limited by input_buffer size
115*/
116UserCallbackFunctionParameters uc_help_(CMD_HELP, uc_help);                   //  uc_help_ has a command string, and function specified
117UserCallbackFunctionParameters uc_settings_(CMD_INPUT_SETTINGS, uc_settings); // uc_settings_ has a command string, and function specified
118// this command will accept seven arguments of the type specified, in order, it will not run the function unless all arguments are valid
119UserCallbackFunctionParameters uc_test_(CMD_TEST, uc_test_input_types,

128void setup()
129{
130  Serial.begin(115200); //  set up Serial
131
132  inputHandler.SetDefaultHandler(uc_unrecognized); // set default function, called when user input has no match or is not valid
133  inputHandler.AddUserCommand(&uc_help_);          // lists commands available to the user
134  inputHandler.AddUserCommand(&uc_settings_);      // lists UserInput class settings
135  inputHandler.AddUserCommand(&uc_test_);          // input type test
136  inputHandler.ListUserCommands();                 //  lists commands available to the user
137}

139void loop()
140{
141  //  inputHandler.ReadCommand(data, len);
142  inputHandler.GetCommandFromStream(Serial);    //  read commands from a stream, hardware or software should work
143  if (inputHandler.OutputIsAvailable() == true) // if there's something to print
144  {
145    Serial.println(output_buffer); // print output_buffer, which is formatted into a string by UserInput's methods
146    string_pos = 0;                //  reset output_buffer's index
147    for (uint16_t i = 0; i < output_buffer_size; ++i)
148    {
149      output_buffer[i] = '\0'; // reinit output_buffer
150    }
151  }
152}

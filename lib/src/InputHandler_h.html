105/**

107 *
108 * these are the different types of user input the process can accept

125    _LAST       ///<  reserved

128/**
129 * @brief InputProcessDelimiterSequences struct holds user defined input data delimiters
130 *
131 * This struct holds information about the delimiter sequences the process will be using.
132 * A delimiter sequence is a predefined number or set of numbers that is used to separate
133 * input arguments, subcommands or data.  The input process needs to know three things
134 * about the delimiter sequences you want to use.  The number of delimiter sequences there are
135 * (up to UI_MAX_DELIM_SEQ), the 8-bit byte (char) length of each delimiter sequence, and
136 * finally each delimiter sequence which can be up to UI_DELIM_SEQ_PGM_LEN in length.
137 */
138struct InputProcessDelimiterSequences
139{
140    size_t num_seq;                                                       ///< the number of token delimiters in delimiter_sequences
141    IH::ui_max_num_delim_seq_t delimiter_lens[UI_MAX_NUM_DELIM_SEQ];      ///< delimiter sequence lens
142    char delimiter_sequences[UI_MAX_NUM_DELIM_SEQ][UI_DELIM_SEQ_PGM_LEN]; ///< string-literal "" delimiter sequence array
143};
144
145/**
146 * @brief InputProcessStartStopSequences struct holds regex-like start-stop match sequence pairs
147 *
148 * This struct holds information about the start-stop sequence pairs the process will be using.
149 * A start-stop sequence is a pair of numbers or a pair of a set of numbers which do not have to
150 * be identical to one another.  The "start" sequence demarcs the beginning of a chunk of data,
151 * the "stop" sequence demarcs the end of the chunk of data. The input process needs to know
152 * three things about the start-stop sequences you want to use.  The number of start-stop
153 * sequences there are (up to UI_MAX_START_STOP_SEQ), the 8-bit byte (char) length of each
154 * start-stop sequence, and finally each start-stop sequence which can be up to
155 * UI_START_STOP_SEQ_PGM_LEN in length.
156 */
157struct InputProcessStartStopSequences
158{
159    size_t num_seq;                                                                       ///< num start/stop sequences
160    IH::ui_max_num_start_stop_seq_t start_stop_sequence_lens[UI_MAX_NUM_START_STOP_SEQ];  ///< start stop sequence lens
161    char start_stop_sequence_pairs[UI_MAX_NUM_START_STOP_SEQ][UI_START_STOP_SEQ_PGM_LEN]; ///< start/stop sequences.  Match start, match end, copy what is between
162};

301     * @brief CommandConstructor Constructor

303     * Linked-list primer: https://www.programiz.com/dsa/linked-list
304     *
305     * These constructors are chained together as a linked-list; this CommandConstructor
306     * object contains a reference CommandConstructor::next_command to the next node
307     * in the linked-list.  The list is linked together in UserInput::addCommand().
308     *
309     * Before using, construct a UserInput object and a CommandParameters object.
310     * @param parameters pointer to parameters struct or array of parameters structs
311     * @param parameter_array_elements number of elements in the parameter array
312     * @param tree_depth depth of command tree
313     */
314    CommandConstructor(const CommandParameters* parameters, const IH::ui_max_commands_in_tree_t parameter_array_elements = 1, const IH::ui_max_tree_depth_per_command_t tree_depth = 0)
315        : prm(parameters), param_array_len(parameter_array_elements), tree_depth(tree_depth + 1U), calc(NULL), next_command(NULL)
316    {
317    }
318    const CommandParameters* prm;                        ///< pointer to PROGMEM CommandParameters array
319    const IH::ui_max_commands_in_tree_t param_array_len; ///< user input param array len, either as digits or through nprms
320    const IH::ui_max_commands_in_tree_t tree_depth;      ///< user input depth + 1
321    CommandRuntimeCalc* calc;                            ///< pointer to CommandRuntimeCalc struct

599    /**
600     * @brief transform 2d matrix indices to flat array index
601     *
602     * use this to access a dynamically allocated array like a 2d matrix,
603     * this is much more performant than looping to allocate a (n>1)d array,
604     * and looping again to free allocated ram.
605     *
606     * @param m_width width of the matrix
607     * @param row row you want to access
608     * @param col column you want to access
609     * @return size_t the transformed index
610     */
611    size_t mIndex(size_t m_width, size_t row, size_t col) const { return row + m_width * col; }

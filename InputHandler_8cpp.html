 35bool UserInput::getToken(char *token_buffer, uint8_t *data, size_t len, uint16_t *data_index)
 36{
 37    bool got_token = false;
 38    char incoming = 0;                   // cast data[data_index] to char and run tests on incoming
 39    bool token_flag[2] = {false, false}; // token state machine, point to a token once
 40    uint32_t data_length = (uint32_t)len;
 41    for (uint16_t i = *data_index; i < data_length; ++i)
 42    {
 43        if (got_token == true)
 44        {
 45#if defined(_DEBUG_USER_INPUT)
 46            if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
 47            {
 48                (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
 49                                                PSTR(">%s $DEBUG: got the token at the beginning of the for loop.\n"), _username_);
 50                _output_flag = true;
 51            }
 52#endif
 53            break;
 54        }
 55        incoming = (char)data[*data_index];
 56#ifdef _DEBUG_USER_INPUT
 57        if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
 58        {
 59            (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
 60                                            PSTR(">%s $DEBUG: incoming char '%c' data_index = %lu.\n"),
 61                                            _username_,
 62                                            incoming,
 63                                            (uint16_t)*data_index);
 64            _output_flag = true;
 65        }
 66#endif
 67        //  remove control characters that are not in a c-string argument, usually CRLF '\r' '\n'
 68        //  replace delimiter
 69        if (iscntrl(incoming) == true || incoming == (char)_delim_[0])
 70        {
 71            token_buffer[*data_index] = *_null_;
 72            token_flag[0] = false;
 73        }
 74        else if (incoming == *_c_str_delim_) // switch logic for c-string input
 75        {
 76            token_buffer[*data_index] = *_null_;              // replace the c-string delimiter
 77            if (((uint16_t)(*data_index) + 1U) < data_length) //  don't need to do this if we're at the end of user input
 78            {
 79                bool point_to_beginning_of_c_string = true; // c-string pointer assignment flag
 80                (*data_index)++;
 81                for (uint16_t j = *data_index; j < data_length; ++j) // this for loop starts at whatever data_index is equal to and has the potential to iterate up to len
 82                {
 83                    incoming = (char)data[*data_index]; // fetch the next incoming char
 84                    if (incoming == *_c_str_delim_)     // if the next incoming char is a '\"'
 85                    {
 86                        token_buffer[*data_index] = *_null_; // replace the c-string delimiter
 87                        (*data_index)++;                     // increment the tokenized string index
 88                        break;
 89                    }
 90                    token_buffer[*data_index] = incoming;       // else assign incoming to token_buffer[data_index]
 91                    if (point_to_beginning_of_c_string == true) // a pointer will be assigned if point_to_beginning_of_c_string == true
 92                    {
 93                        data_pointers[data_pointers_index] = &token_buffer[*data_index]; // point to this position in token_buffer
 94                        data_pointers_index++;                                           //  increment pointer index
 95                        point_to_beginning_of_c_string = false;                          //  only set one pointer per c-string
 96                        got_token = true;
 97#ifdef _DEBUG_USER_INPUT
 98                        if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
 99                        {
100                            (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
101                                                            PSTR(">%s $DEBUG: got the c-string token.\n"), _username_);
102                            _output_flag = true;
103                        }
104#endif
105                    }
106                    (*data_index)++; // increment the tokenized string index
107                }
108            }
109        }
110        else // this is a non c-string token
111        {
112            if (incoming == '\\' && (*data_index < data_length))
113            {
114                token_buffer[*data_index] = '\0';
115                (*data_index)++; // increment buffer index
116                incoming = combineControlCharacters((char)data[*data_index]);
117            }
118            token_buffer[*data_index] = incoming; // assign incoming to token_buffer at data_index
119            token_flag[0] = true;                 // set token available sentinel to true
120        }
121
122        if (token_flag[0] != token_flag[1]) // if the token state has changed
123        {
124            if (token_flag[0] == true) // if there's a new token
125            {
126                data_pointers[data_pointers_index] = &token_buffer[*data_index]; // assign a pointer the beginning of it
127                data_pointers_index++;                                           // and increment the pointer index
128            }
129            else
130            {
131#ifdef _DEBUG_USER_INPUT
132                if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
133                {
134                    (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
135                                                    PSTR(">%s $DEBUG: got the token token_flag[0] == false.\n"), _username_);
136                    _output_flag = true;
137                }
138#endif
139                got_token = true;
140            }
141            token_flag[1] = token_flag[0]; // track the state
142        }
143        if (token_flag[0] == true && (uint16_t)*data_index == (data_length - 1U))
144        {
145#ifdef _DEBUG_USER_INPUT
146            if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
147            {
148                (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
149                                                PSTR(">%s $DEBUG: got the token token_flag[0] == true && data_index == len - 1.\n"), _username_);
150                _output_flag = true;
151            }
152#endif
153            got_token = true;
154        }
155
156        if (*data_index < data_length) // if we are at the end of input data
157        {
158            (*data_index)++; // increment buffer index
159        }
160    }
161    return got_token;
162}

164bool UserInput::validateUserInput(UserCallbackFunctionParameters *cmd, uint8_t arg_type, uint16_t data_pointers_index)
165{
166    uint16_t strlen_data = strlen(data_pointers[data_pointers_index]);
167    bool found_negative_sign = ((char)data_pointers[data_pointers_index][0] == *_negative_sign) ? true : false;
168
169    if (arg_type < (size_t)_UITYPE::CHAR)
170    {
171        // for unsigned integers
172        if (arg_type < (size_t)_UITYPE::INT16_T)
173        {
174            if (found_negative_sign == true)
175            {
176                return false;
177            }
178            for (uint16_t j = 0; j < strlen_data; ++j)
179            {
180                if (isdigit(data_pointers[data_pointers_index][j]) == false)
181                {
182                    return false;
183                }
184            }
185        }
186        // for integer numbers
187        if (arg_type == (size_t)_UITYPE::INT16_T)
188        {
189            if (found_negative_sign == true) //  negative
190            {
191                for (uint16_t j = 1; j < (strlen_data - 1); ++j)
192                {
193                    if (isdigit(data_pointers[data_pointers_index][j]) == false)
194                    {
195                        return false;
196                    }
197                }
198            }
199            else //  positive
200            {
201                for (uint16_t j = 0; j < strlen_data; ++j)
202                {
203                    if (isdigit(data_pointers[data_pointers_index][j]) == false)
204                    {
205                        return false;
206                    }
207                }
208            }
209        }
210        // for floating point numbers
211        if (arg_type == (size_t)_UITYPE::FLOAT)
212        {
213            uint8_t found_dot = 0;
214            uint8_t num_digits = 0;
215            if (found_negative_sign == true) //  negative
216            {
217                uint8_t not_digits = 0;
218                /*
219                    we already know there is a '-' at data[i + 1][0] because found_negative_sign is set
220                    so start the for loop at an index of one
221                */
222                for (uint16_t j = 1; j < strlen_data; ++j)
223                {
224                    if (data_pointers[data_pointers_index][j] == *_dot)
225                    {
226                        found_dot++;
227                    }
228                    if (isdigit(data_pointers[data_pointers_index][j]) == true)
229                    {
230                        num_digits++;
231                    }
232                    else
233                    {
234                        not_digits++;
235                    }
236                }
237                if (found_dot == 0) // if there is no fraction
238                {
239                    found_dot++;
240                }
241                /*
242                    if we find one '.' and one '-' or one '-' and no '.' increment num_digits to include the sign, else type match will fail
243                    this is to allow negative numbers with no decimal or fraction like "-5" and also input like "-5.0"
244                */
245                if ((not_digits == 2 && found_dot == 1) || (not_digits == 1 && found_dot == 1))
246                {
247                    num_digits++; // count the negative sign
248                }
249            }
250            else //  positive
251            {
252                for (uint16_t j = 0; j < strlen_data; ++j)
253                {
254                    if (data_pointers[data_pointers_index][j] == *_dot)
255                    {
256                        found_dot++;
257                    }
258                    if (isdigit(data_pointers[data_pointers_index][j]) == true)
259                    {
260                        num_digits++;
261                    }
262                }
263            }
264            if (found_dot > 1 || (num_digits + found_dot) != strlen_data)
265            {
266                return false;
267            }
268        }
269    }
270    /*
271        For char and c-string input.
272        Types allowed are printable characters, punctuation, control characters \r\n etc, and digits 0-9
273    */
274    else if (arg_type == (size_t)_UITYPE::CHAR)
275    {
276        if (strlen_data > 1)
277        {
278            return false;
279        }
280        if (isprint(data_pointers[data_pointers_index][0]) ||
281            ispunct(data_pointers[data_pointers_index][0]) ||
282            iscntrl(data_pointers[data_pointers_index][0]) ||
283            isdigit(data_pointers[data_pointers_index][0]))
284        {
285        }
286        else
287        {
288            return false;
289        }
290    }
291    else if (arg_type == (size_t)_UITYPE::C_STRING)
292    {
293        for (uint16_t j = 0; j < strlen_data; ++j)
294        {
295            if (isprint(data_pointers[data_pointers_index][j]) ||
296                ispunct(data_pointers[data_pointers_index][j]) ||
297                iscntrl(data_pointers[data_pointers_index][j]) ||
298                isdigit(data_pointers[data_pointers_index][j]))
299            {
300            }
301            else
302            {
303                return false;
304            }
305        }
306    }
307    else //  unknown types always return false
308    {
309        return false;
310    }
311
312    return true;
313}

315void UserInput::launchFunction(UserCallbackFunctionParameters *cmd)
316{
317    if (UserInput::OutputIsEnabled())
318    {
319        (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len, PSTR(">%s $%s"), _username_, data_pointers[0]);
320        for (uint16_t i = 0; i < rec_num_arg_strings; ++i)
321        {
322            if (iscntrl(data_pointers[i][0]))
323            {
324                char temp_buffer[13] = {'\0'};
325                UserInput::escapeCharactersSoTheyPrint(data_pointers[i + 1], temp_buffer);
326                (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len, PSTR(" %s"), temp_buffer);
327            }
328            else
329            {
330                (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len, PSTR(" %s"), data_pointers[i + 1]);
331            }
332        }
333        (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len, PSTR("\n"));
334        _output_flag = true;
335    }
336    data_pointers_index = 0;
337    cmd->function(this);
338}

340void UserInput::ReadCommand(uint8_t *data, size_t len)
341{
342    if (UserInput::OutputIsEnabled() && len > USER_INPUT_MAX_INPUT_LENGTH)
343    {
344        (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
345                                        PSTR(">%s $ERROR: user input exceeds max allowed input length.\n"), _username_);
346        _output_flag = true;
347        return;
348    }
349    uint16_t data_index = 0; // data iterator
350
351    //  should maybe see if there is enough memory to allocate the token buffer
352    token_buffer = new char[len + 1](); // place to chop up the input
353    data_pointers_index = 0;            // token buffer pointers
354    rec_num_arg_strings = 0;            // number of tokens read from data
355    bool match = false;
356    UserCallbackFunctionParameters *cmd;
357
358    /*
359        this tokenizes an input buffer, it should work with any 8 bit input type that represents char
360        char tokenized_string[] = "A\0Tokenized\0C-string\0"
361        char non_tokenized_string[] = "A Non Tokenized C-string" <-- still has a \0 at the end of the string to terminate it
362    */
363    if (getToken(token_buffer, data, len, &data_index) == true) // if there was a token
364    {
365        bool input_type_match_flag[USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS] = {false};
366        bool all_arguments_valid = true;                                                      // error sentinel
367        bool command_matched = false;                                                         // error sentinel
368        for (cmd = commands_head_; cmd != NULL; cmd = cmd->next_callback_function_parameters) // iterate through user commands
369        {
370            if (strcmp(data_pointers[0], cmd->command) == 0) // match
371            {
372                command_matched = true;
373                if (cmd->num_args == 0) // command with no arguments
374                {
375                    while (getToken(token_buffer, data, len, &data_index) == true && rec_num_arg_strings < USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS)
376                    {
377                        rec_num_arg_strings++;
378                        if (rec_num_arg_strings > 0)
379                        {
380                            break;
381                        }
382                    }
383#ifdef _DEBUG_USER_INPUT
384                    if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
385                    {
386                        (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
387                                                        PSTR(">%s $DEBUG: match zero argument command <%s>.\n"),
388                                                        _username_,
389                                                        cmd->command);
390                        _output_flag = true;
391                    }
392#endif
393                    match = true;        // don't run default callback
394                    launchFunction(cmd); // launch the matched command
395                    break;               // break out of the for loop
396                }
397                else if (cmd->num_args > 0) // cmd->num_args > 0
398                {
399                    while (getToken(token_buffer, data, len, &data_index) == true && rec_num_arg_strings < USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS)
400                    {
401                        input_type_match_flag[rec_num_arg_strings] = validateUserInput(cmd, UI_PGM_READ_BYTE(UI_DEREFERENCE cmd->_arg_type[rec_num_arg_strings]), data_pointers_index - 1); // validate the token
402                        if (input_type_match_flag[rec_num_arg_strings] == false)                                                                                                            // if the token was not valid input
403                        {
404                            all_arguments_valid = false; // set the error sentinel to true
405                        }
406                        rec_num_arg_strings++;
407                    }
408                    if (rec_num_arg_strings == cmd->num_args && all_arguments_valid == true) //  if we received the expected amount of arguments and all of them are valid
409                    {
410#ifdef _DEBUG_USER_INPUT
411                        if (UserInput::OutputIsEnabled() && EnableDebugOutput == true)
412                        {
413                            (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
414                                                            PSTR(">%s $DEBUG: match command <%s>.\n"),
415                                                            _username_,
416                                                            cmd->command);
417                            _output_flag = true;
418                        }
419#endif
420                        match = true;        // don't run default callback
421                        launchFunction(cmd); // launch the matched command
422                    }
423                }
424                break;
425            }
426        }
427        if (!match && default_handler_ != NULL) // if there was no command match and a default handler is configured
428        {
429            // format a string with useful information
430            if (UserInput::OutputIsEnabled())
431            {
432                char data_char_buffer[len + 1];
433                data_char_buffer[len] = '\0';
434                for (uint16_t i = 0; i < len; ++i)
435                {
436                    data_char_buffer[i] = (char)data[i];
437                }
438                (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
439                                                PSTR(">%s $ERROR: %s\n"),
440                                                _username_,
441                                                data_char_buffer);
442                _output_flag = true;
443                if (!command_matched)
444                {
445                    (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
446                                                    PSTR("Command <%s> unknown.\n"),
447                                                    data_pointers[0]);
448                    _output_flag = true;
449                }
450                if (command_matched && all_arguments_valid == false)
451                {
452                    uint16_t err_n_args = (cmd->num_args > rec_num_arg_strings) ? rec_num_arg_strings : cmd->num_args;
453                    for (uint8_t i = 0; i < err_n_args; ++i)
454                    {
455                        if (input_type_match_flag[i] == false)
456                        {
457                            (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
458                                                            PSTR("<%s> argument <%u>: should be %s; received <%s>.\n"),
459                                                            cmd->command, i + 1,
460                                                            (char *)UI_PGM_READ_DWORD(UI_DEREFERENCE(_input_type_strings[uint8_t(cmd->_arg_type[i])])),
461                                                            data_pointers[i + 1]);
462                        }
463                        _output_flag = true;
464                    }
465                }
466                if (command_matched && (rec_num_arg_strings != cmd->num_args))
467                {
468                    (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
469                                                    PSTR("<%s> received <%02u> arguments; <%s> expects <%02u> arguments.\n"),
470                                                    cmd->command, (rec_num_arg_strings), cmd->command, cmd->num_args);
471                    _output_flag = true;
472                }
473            }
474            (*default_handler_)(this); // run the default handler
475        }
476    }
477    else
478    {
479        if (UserInput::OutputIsEnabled() && len > 0)
480        {
481            (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
482                                            PSTR(">%s $ERROR: No tokens retrieved.\n"), _username_);
483            _output_flag = true;
484        }
485    }
486    delete[] token_buffer;
487}

489void UserInput::GetCommandFromStream(Stream &stream, uint16_t rx_buffer_size, const char *end_of_line_char)
490{
491    if (stream_buffer_allocated == false)
492    {
493        stream_data = new uint8_t[rx_buffer_size]; // an array to store the received data
494        stream_buffer_allocated = true;
495    }
496    char *rc = (char *)stream_data;
497
498    while (stream.available() > 0 && new_stream_data == false)
499    {
500        rc[stream_data_index] = stream.read();
501        if (rc[stream_data_index] == _term_[0] || rc[stream_data_index] == _term_[1])
502        {
503            stream_data[stream_data_index] = '\0';
504            new_stream_data = true;
505        }
506        else if (stream_data_index < (rx_buffer_size - 1))
507        {
508            stream_data_index++;
509        }
510    }
511    if (new_stream_data == true)
512    {
513        UserInput::ReadCommand(stream_data, stream_data_index);
514        stream_data_index = 0;
515        new_stream_data = false;
516        delete[] stream_data;
517        stream_buffer_allocated = false;
518    }
519}

539void UserInput::escapeCharactersSoTheyPrint(const char *input, char *output)
540{
541    uint16_t len = strlen(input);
542    for (uint16_t i = 0; i < len; ++i)
543    {
544        switch ((char)input[i])
545        {
546        case '\a':
547            strcat_P(output, PSTR("\\a"));
548            break;
549        case '\b':
550            strcat_P(output, PSTR("\\b"));
551            break;
552        case '\f':
553            strcat_P(output, PSTR("\\f"));
554            break;
555        case '\n':
556            strcat_P(output, PSTR("\\n"));
557            break;
558        case '\r':
559            strcat_P(output, PSTR("\\r"));
560            break;
561        case '\t':
562            strcat_P(output, PSTR("\\t"));
563            break;
564        case '\v':
565            strcat_P(output, PSTR("\\v"));
566            break;
567        case '\"':
568            strcat_P(output, PSTR("\\\""));
569            break;
570        case ' ':
571            strcat_P(output, PSTR(" "));
572        default:
573            if (i == len)
574                strcat(output, input);
575            break;
576        }
577    }
578}

580char UserInput::combineControlCharacters(char input)
581{
582    switch ((char)input)
583    {
584    case 'a':
585        return (char)'\a';
586    case 'b':
587        return (char)'\b';
588    case 'f':
589        return (char)'\f';
590    case 'n':
591        return (char)'\n';
592    case 'r':
593        return (char)'\r';
594    case 't':
595        return (char)'\t';
596    case 'v':
597        return (char)'\v';
598    case '"':
599        return (char)'\"';
600    default:
601        return input;
602    }

610        inputprocess->escapeCharactersSoTheyPrint(_term_, temp_settings[0]);
611        inputprocess->escapeCharactersSoTheyPrint(_delim_, temp_settings[1]);
612        inputprocess->escapeCharactersSoTheyPrint(_c_str_delim_, temp_settings[2]);
613        (*_string_pos) += UI_SNPRINTF_P(_output_buffer + (*_string_pos), _output_buffer_len,
614                                        PSTR("username = \"%s\"\n"
615                                             "end_of_line_characters = \"%s\"\n"
616                                             "token_delimiter = \"%s\"\n"
617                                             "c_string_delimiter = \"%s\"\n"),
618                                        inputprocess->_username_,
619                                        (char *)temp_settings[0],
620                                        (char *)temp_settings[1],
621                                        (char *)temp_settings[2]);
622        _output_flag = true;
623    }
624}

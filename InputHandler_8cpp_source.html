21{

52bool UserInput::getToken(char *token_buffer, uint8_t *data, size_t len, size_t *data_index)
53{
54 bool got_token = false;
55 char incoming = 0; // cast data[data_index] to char and run tests on incoming
56 bool token_flag[2] = {false, false}; // token state machine, point to a token once
57 uint32_t data_length = (uint32_t)len;
58 for (uint16_t i = *data_index; i < data_length; ++i)
59 {
60 if (got_token == true)
61 {
62 #if defined(_DEBUG_USER_INPUT)
63 if (UserInput::OutputIsEnabled())
64 {
65 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
66 PSTR(">%s $DEBUG: got the token at the beginning of the for loop.\n"),
67 _username_);
68 _output_flag = true;
69 }
70 #endif
71 break;
72 }
73 incoming = (char)data[*data_index];
74 #if defined(_DEBUG_USER_INPUT)
75 if (UserInput::OutputIsEnabled())
76 {
77 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
78 PSTR(">%s $DEBUG: incoming char '%c' data_index = %lu.\n"),
79 _username_,
80 incoming,
81 (uint16_t)*data_index);
82 _output_flag = true;
83 }
84 #endif
85 // remove control characters that are not in a c-string argument, usually CRLF '\r' '\n'
86 // replace delimiter
87 if (iscntrl(incoming) == true || incoming == (char)_delim_[0])
88 {
89 token_buffer[*data_index] = *_null_;
90 token_flag[0] = false;
91 }
92 else if (incoming == *_c_str_delim_) // switch logic for c-string input
93 {
94 token_buffer[*data_index] = *_null_; // replace the c-string delimiter
95 if (((uint16_t)(*data_index) + 1U) < data_length) // don't need to do this if we're at the end of user input
96 {
97 bool point_to_beginning_of_c_string = true; // c-string pointer assignment flag
98 (*data_index)++;
99 for (uint16_t j = *data_index; j < data_length; ++j) // this for loop starts at whatever data_index is equal to and has the potential to iterate up to len
100 {
101 incoming = (char)data[*data_index]; // fetch the next incoming char
102 if (incoming == *_c_str_delim_) // if the next incoming char is a '\"'
103 {
104 token_buffer[*data_index] = *_null_; // replace the c-string delimiter
105 (*data_index)++; // increment the tokenized string index
106 break;
107 }
108 token_buffer[*data_index] = incoming; // else assign incoming to token_buffer[data_index]
109 if (point_to_beginning_of_c_string == true) // a pointer will be assigned if point_to_beginning_of_c_string == true
110 {
111 data_pointers[data_pointers_index] = &token_buffer[*data_index]; // point to this position in token_buffer
112 data_pointers_index++; // increment pointer index
113 point_to_beginning_of_c_string = false; // only set one pointer per c-string
114 got_token = true;
115 #if defined(_DEBUG_USER_INPUT)
116 if (UserInput::OutputIsEnabled())
117 {
118 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
119 PSTR(">%s $DEBUG: got the c-string token.\n"),
120 _username_);
121 _output_flag = true;
122 }
123 #endif
124 }
125 (*data_index)++; // increment the tokenized string index
126 }
127 }
128 }
129 else // this is a non c-string token
130 {
131 if (incoming == '\\' && (*data_index < data_length))
132 {
133 token_buffer[*data_index] = '\0';
134 (*data_index)++; // increment buffer index
135 incoming = combineControlCharacters((char)data[*data_index]);
136 }
137 token_buffer[*data_index] = incoming; // assign incoming to token_buffer at data_index
138 token_flag[0] = true; // set token available sentinel to true
139 }
140
141 if (token_flag[0] != token_flag[1]) // if the token state has changed
142 {
143 if (token_flag[0] == true) // if there's a new token
144 {
145 data_pointers[data_pointers_index] = &token_buffer[*data_index]; // assign a pointer the beginning of it
146 data_pointers_index++; // and increment the pointer index
147 }
148 else
149 {
150 #if defined(_DEBUG_USER_INPUT)
151 if (UserInput::OutputIsEnabled())
152 {
153 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
154 PSTR(">%s $DEBUG: got the token token_flag[0] == false.\n"),
155 _username_);
156 _output_flag = true;
157 }
158 #endif
159 got_token = true;
160 }
161 token_flag[1] = token_flag[0]; // track the state
162 }
163 if (token_flag[0] == true && (uint16_t)*data_index == (data_length - 1U))
164 {
165 #if defined(_DEBUG_USER_INPUT)
166 if (UserInput::OutputIsEnabled())
167 {
168 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
169 PSTR(">%s $DEBUG: got the token token_flag[0] == true && data_index == len - 1.\n"),
170 _username_);
171 _output_flag = true;
172 }
173 #endif
174 got_token = true;
175 }
176
177 if (*data_index < data_length) // if we are at the end of input data
178 {
179 (*data_index)++; // increment buffer index
180 }
181 }
182 return got_token;
183}

185bool UserInput::validateUserInput(uint8_t arg_type, size_t data_pointers_index)
186{
187 uint16_t strlen_data = strlen(data_pointers[data_pointers_index]);
188 bool found_negative_sign = ((char)data_pointers[data_pointers_index][0] == *((PGM_P)pgm_read_ptr(&(ui_defaults_progmem_ptr[neg_e])))) ? true : false;
189
190 if (arg_type < (size_t)UITYPE::CHAR)
191 {
192 // for unsigned integers
193 if (arg_type < (size_t)UITYPE::INT16_T)
194 {
195 if (found_negative_sign == true)
196 {
197 return false;
198 }
199 for (uint16_t j = 0; j < strlen_data; ++j)
200 {
201 if (isdigit(data_pointers[data_pointers_index][j]) == false)
202 {
203 return false;
204 }
205 }
206 }
207 // for integer numbers
208 if (arg_type == (size_t)UITYPE::INT16_T)
209 {
210 if (found_negative_sign == true) // negative
211 {
212 for (uint16_t j = 1; j < (strlen_data - 1); ++j)
213 {
214 if (isdigit(data_pointers[data_pointers_index][j]) == false)
215 {
216 return false;
217 }
218 }
219 }
220 else // positive
221 {
222 for (uint16_t j = 0; j < strlen_data; ++j)
223 {
224 if (isdigit(data_pointers[data_pointers_index][j]) == false)
225 {
226 return false;
227 }
228 }
229 }
230 }
231 // for floating point numbers
232 if (arg_type == (size_t)UITYPE::FLOAT)
233 {
234 uint8_t found_dot = 0;
235 uint8_t num_digits = 0;
236 if (found_negative_sign == true) // negative
237 {
238 uint8_t not_digits = 0;
239 /*
240 we already know there is a '-' at data[i + 1][0] because found_negative_sign is set
241 so start the for loop at an index of one
242 */
243 for (uint16_t j = 1; j < strlen_data; ++j)
244 {
245 if (data_pointers[data_pointers_index][j] == *((PGM_P)pgm_read_ptr(&ui_defaults_progmem_ptr[dot_e])))
246 {
247 found_dot++;
248 }
249 if (isdigit(data_pointers[data_pointers_index][j]) == true)
250 {
251 num_digits++;
252 }
253 else
254 {
255 not_digits++;
256 }
257 }
258 if (found_dot == 0) // if there is no fraction
259 {
260 found_dot++;
261 }
262 /*
263 if we find one '.' and one '-' or one '-' and no '.' increment num_digits to include the sign, else type match will fail
264 this is to allow negative numbers with no decimal or fraction like "-5" and also input like "-5.0"
265 */
266 if ((not_digits == 2 && found_dot == 1) || (not_digits == 1 && found_dot == 1))
267 {
268 num_digits++; // count the negative sign
269 }
270 }
271 else // positive
272 {
273 for (uint16_t j = 0; j < strlen_data; ++j)
274 {
275 if (data_pointers[data_pointers_index][j] == *((PGM_P)pgm_read_ptr(&ui_defaults_progmem_ptr[dot_e])))
276 {
277 found_dot++;
278 }
279 if (isdigit(data_pointers[data_pointers_index][j]) == true)
280 {
281 num_digits++;
282 }
283 }
284 }
285 if (found_dot > 1 || (num_digits + found_dot) != strlen_data)
286 {
287 return false;
288 }
289 }
290 }
291 /*
292 For char and c-string input.
293 Types allowed are printable characters, punctuation, control characters \r\n etc, and digits 0-9
294 */
295 else if (arg_type == (size_t)UITYPE::CHAR)
296 {
297 if (strlen_data > 1)
298 {
299 return false;
300 }
301 if (isprint(data_pointers[data_pointers_index][0]) ||
302 ispunct(data_pointers[data_pointers_index][0]) ||
303 iscntrl(data_pointers[data_pointers_index][0]) ||
304 isdigit(data_pointers[data_pointers_index][0]))
305 {
306 }
307 else
308 {
309 return false;
310 }
311 }
312 else if (arg_type == (size_t)UITYPE::C_STRING)
313 {
314 for (uint16_t j = 0; j < strlen_data; ++j)
315 {
316 if (isprint(data_pointers[data_pointers_index][j]) ||
317 ispunct(data_pointers[data_pointers_index][j]) ||
318 iscntrl(data_pointers[data_pointers_index][j]) ||
319 isdigit(data_pointers[data_pointers_index][j]))
320 {
321 }
322 else
323 {
324 return false;
325 }
326 }
327 }
328 else if (arg_type == (size_t)UITYPE::NOTYPE)
329 {
330 // no type validation performed
331 return true;
332 }
333 else // unknown types always return false
334 {
335 return false;
336 }
337
338 return true;
339}

342{

344 {
345 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len, PSTR(">%s $%s"),

362 _output_flag = true;
363 }

368void UserInput::ReadCommandFromBuffer(uint8_t* data, size_t len)
369{
370 if (len > USER_INPUT_MAX_INPUT_LENGTH)
371 {
372 if (UserInput::OutputIsEnabled())
373 {
374 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
375 PSTR(">%s $ERROR: input is greater than USER_INPUT_MAX_INPUT_LENGTH.\n"),
376 _username_);
377 _output_flag = true;
378 }
379 return;
380 }
381
382 token_buffer = new char[len + 1](); // place to chop up the input
383 if (token_buffer == nullptr) // if there was an error allocating the memory
384 {
385 if (UserInput::OutputIsEnabled())
386 {
387 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
388 PSTR(">%s $ERROR: not enough free ram to allocate the token buffer.\n"),
389 _username_);
390 _output_flag = true;
391 }
392 return;
393 }
394
395 size_t data_index = 0; // data iterator
396 data_pointers_index = 0; // token buffer pointers
397 rec_num_arg_strings = 0; // number of tokens read from data
398 bool match = false; // command string match
399 bool command_matched = false; // error sentinel
400 UserCommandParameters* cmd; // command parameters pointer
401
402 /*
403 this tokenizes an input buffer, it should work with any 8 bit input type that represents char
404 char tokenized_string[] = "A\0Tokenized\0C-string\0"
405 char non_tokenized_string[] = "A Non Tokenized C-string" <-- still has a \0 at the end of the string to terminate it
406 */
407 if (getToken(token_buffer, data, len, &data_index) == true) // if there was a token
408 {
409 bool input_type_match_flag[USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS] = {false};
410 bool all_arguments_valid = true; // error sentinel
411 for (cmd = commands_head_; cmd != NULL; cmd = cmd->next_command_parameters) // iterate through user commands
412 {
413 if (strcmp(data_pointers[0], cmd->command) == 0) // match
414 {
415 command_matched = true;
416 if (cmd->num_args == 0) // command with no arguments
417 {
418 while (getToken(token_buffer, data, len, &data_index) == true && rec_num_arg_strings < USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS)
419 {
420 rec_num_arg_strings++;
421 if (rec_num_arg_strings > 0)
422 {
423 break;
424 }
425 }
426 #if defined(_DEBUG_USER_INPUT)
427 if (UserInput::OutputIsEnabled())
428 {
429 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
430 PSTR(">%s $DEBUG: match zero argument command <%s>.\n"),
431 _username_,
432 cmd->command);
433 _output_flag = true;
434 }
435 #endif
436 match = true; // don't run default callback
437 launchFunction(cmd); // launch the matched command
438 break; // break out of the for loop
439 }
440 else if (cmd->num_args > 0) // cmd->num_args > 0
441 {
442 while (getToken(token_buffer, data, len, &data_index) == true && rec_num_arg_strings < USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS)
443 {
444 input_type_match_flag[rec_num_arg_strings] = validateUserInput(UserInput::getArgType(cmd, rec_num_arg_strings), data_pointers_index - 1); // validate the token
445 if (input_type_match_flag[rec_num_arg_strings] == false) // if the token was not valid input
446 {
447 all_arguments_valid = false; // set the error sentinel to true
448 }
449 rec_num_arg_strings++;
450 }
451 if (rec_num_arg_strings == cmd->num_args && all_arguments_valid == true) // if we received the expected amount of arguments and all of them are valid
452 {
453 #if defined(_DEBUG_USER_INPUT)
454 if (UserInput::OutputIsEnabled())
455 {
456 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
457 PSTR(">%s $DEBUG: match command <%s>.\n"),
458 _username_,
459 cmd->command);
460 _output_flag = true;
461 }
462 #endif
463 match = true; // don't run default callback
464 launchFunction(cmd); // launch the matched command
465 }
466 }
467 break;
468 }
469 }
470 if (!match && default_function_ != NULL) // if there was no command match and a default function is configured
471 {
472 // format a string with useful information
473 if (UserInput::OutputIsEnabled())
474 {
475 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
476 PSTR(">%s $Invalid input: %s "),
477 _username_,
478 data_pointers[0]);
479 if (command_matched == true)
480 {
481 uint16_t err_n_args = (cmd->num_args > rec_num_arg_strings) ? rec_num_arg_strings : cmd->num_args;
482 if (err_n_args > 0)
483 {
484 for (uint16_t i = 0; i < err_n_args; ++i)
485 {
486 if (input_type_match_flag[i] == false)
487 {
488 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
489 PSTR("%s* "),
490 data_pointers[i + 1]);
491 }
492 else
493 {
494 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
495 PSTR("%s "),
496 data_pointers[i + 1]);
497 }
498 }
499 }
500 }
501 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
502 PSTR("\n"));
503 if (!command_matched)
504 {
505 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
506 PSTR("Command <%s> unknown.\n"),
507 data_pointers[0]);
508 }
509 if (command_matched && all_arguments_valid == false)
510 {
511 uint16_t err_n_args = (cmd->num_args > rec_num_arg_strings) ? rec_num_arg_strings : cmd->num_args;
512 for (uint8_t i = 0; i < err_n_args; ++i)
513 {
514 if (input_type_match_flag[i] == false)
515 {
516 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
517 PSTR(" > arg(%u) should be %s; received \"%s\".\n"),
518 i + 1,
519 (char*)UI_PGM_READ_DWORD(UI_DEREFERENCE(ui_input_type_strings[UserInput::getArgType(cmd, i)])),
520 data_pointers[i + 1]);
521 }
522 }
523 }
524 if (command_matched && (rec_num_arg_strings != cmd->num_args))
525 {
526 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
527 PSTR(" command \"%s\" received <%02u> arguments; %s expects <%02u> arguments.\n"),
528 cmd->command, (rec_num_arg_strings), cmd->command, cmd->num_args);
529 }
530 _output_flag = true;
531 }
532 (*default_function_)(this); // run the default function
533 }
534 }
535 else
536 {
537 if (UserInput::OutputIsEnabled() && len > 0)
538 {
539 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
540 PSTR(">%s $ERROR: No tokens retrieved.\n"),
541 _username_);
542 _output_flag = true;
543 }
544 }
545 delete[] token_buffer;
546}

548void UserInput::GetCommandFromStream(Stream &stream, size_t rx_buffer_size)
549{
550 if (stream_buffer_allocated == false)
551 {
552 stream_data = new uint8_t[rx_buffer_size]; // an array to store the received data
553 if (stream_data == nullptr) // if there was an error allocating the memory
554 {
555 if (UserInput::OutputIsEnabled())
556 {
557 _string_pos += UI_SNPRINTF_P(_output_buffer + _string_pos, _output_buffer_len,
558 PSTR(">%s $ERROR: not enough memory for stream rx buffer\n"),
559 _username_);
560 _output_flag = true;
561 }
562 return;
563 }
564 stream_buffer_allocated = true;
565 }
566 char *rc = (char *)stream_data;
567
568 while (stream.available() > 0 && new_stream_data == false)
569 {
570 rc[stream_data_index] = stream.read();
571 if (rc[stream_data_index] == _term_[0] || rc[stream_data_index] == _term_[1])
572 {
573 stream_data[stream_data_index] = '\0';
574 new_stream_data = true;
575 }
576 else if (stream_data_index < (rx_buffer_size - 1))
577 {
578 stream_data_index++;
579 }
580 }
581 if (new_stream_data == true)
582 {
583 UserInput::ReadCommandFromBuffer(stream_data, stream_data_index);
584 stream_data_index = 0;
585 new_stream_data = false;
586 delete[] stream_data;
587 stream_buffer_allocated = false;
588 }
589}

609void UserInput::escapeCharactersSoTheyPrint(const char *input, char *output)
610{
611 uint16_t len = strlen(input);
612 for (uint16_t i = 0; i < len; ++i)
613 {
614 switch ((char)input[i])
615 {
616 case '\a':
617 strcat_P(output, PSTR("\\a"));
618 break;
619 case '\b':
620 strcat_P(output, PSTR("\\b"));
621 break;
622 case '\f':
623 strcat_P(output, PSTR("\\f"));
624 break;
625 case '\n':
626 strcat_P(output, PSTR("\\n"));
627 break;
628 case '\r':
629 strcat_P(output, PSTR("\\r"));
630 break;
631 case '\t':
632 strcat_P(output, PSTR("\\t"));
633 break;
634 case '\v':
635 strcat_P(output, PSTR("\\v"));
636 break;
637 case '\"':
638 strcat_P(output, PSTR("\\\""));
639 break;
640 default:
641 if (i == len)
642 strcat(output, input);
643 break;
644 }
645 }
646}

648char UserInput::combineControlCharacters(char input)
649{
650 switch ((char)input)
651 {
652 case 'a':
653 return (char)'\a';
654 case 'b':
655 return (char)'\b';
656 case 'f':
657 return (char)'\f';
658 case 'n':
659 return (char)'\n';
660 case 'r':
661 return (char)'\r';
662 case 't':
663 return (char)'\t';
664 case 'v':
665 return (char)'\v';
666 case '"':
667 return (char)'\"';
668 default:
669 return input;
670 }

56 void UserInput::ReadCommandFromBuffer(uint8_t *data, size_t len)
57 {
58  // error checking
59  if (len > (UI_MAX_IN_LEN - 2)) // 65535 - 1(index align) - 1(space for null '\0')
60  {
61  _ui_out(PSTR(">%s $ERROR: input is too long.\n"), _username_);
62  return;
63  }
64 
65  // this is declared here to test if token_buffer == nullptr (error condition)
66  token_buffer = new char[len + 1](); // place to chop up the input
67  if (token_buffer == nullptr) // if there was an error allocating the memory
68  {
69  _ui_out(PSTR(">%s $ERROR: cannot allocate ram for token_buffer.\n"), _username_);
70  return;
71  }
72  // end error checking
73 
74  // reinit data_pointers
75  for (size_t i = 0; i < (UI_MAX_ARGS + 1); ++i)
76  {
77  data_pointers[i] = NULL;
78  }
79 
80  uint8_t tokens_received = 0; // amount of delimiter separated tokens
81  size_t data_index = 0; // data iterator
82  data_pointers_index = 0; // token buffer pointers
83  rec_num_arg_strings = 0; // number of tokens read from data
84  bool match = false; // command string match
85  bool command_matched = false; // error sentinel
86  CommandConstructor *cmd; // command parameters pointer
87  Parameters prm; // Parameters struct
88 
89  /*
90  this tokenizes an input buffer, it should work with any 8 bit input type that represents char
91  char tokenized_string[] = "A\0Tokenized\0C-string\0"
92  char non_tokenized_string[] = "A Non Tokenized C-string" <-- still has a \0 at the end of the string to terminate it
93  */
94  while (true)
95  {
96  if (getToken(token_buffer, data, len, &data_index) == true)
97  {
98  tokens_received++; // increment tokens_received
99  if (tokens_received == (UI_MAX_ARGS + 1)) // index sentinel
100  {
101  break;
102  }
103  }
104  else
105  {
106  break;
107  }
108  }
109  data_pointers_index_max = tokens_received;
110  // error condition
111  if (tokens_received == 0)
112  {
113  _ui_out(PSTR(">%s $ERROR: No tokens retrieved.\n"), _username_);
114  delete[] token_buffer;
115  return;
116  }
117  // end error condition
118 
119  bool input_type_match_flag[UI_MAX_ARGS] = {false};
120  bool all_arguments_valid = true; // error sentinel
121 
122  for (cmd = commands_head_; cmd != NULL; cmd = cmd->next_command_parameters) // iterate through user commands
123  {
124  // &(cmd->prm[0]) points to the first parameters struct for this cmd
125  memcpy_P(&prm, &(cmd->prm[0]), sizeof(prm)); // move working variables to ram
126  // if the first test is false, the strcmp test is not evaluated
127  if (strcmp(data_pointers[0], prm.command) == 0) // match root command
128  {
129  _current_search_depth = 1; // start searching for subcommands at depth 1
130  data_pointers_index = 1; // index 1 of data_pointers is the token after the root command
131  command_matched = true; // root command match flag
132  failed_on_subcommand = 0; // subcommand error index
133  bool subcommand_matched = false; // subcommand match flag
134  uint8_t prm_idx = 0;
135  // see if command has any subcommands, validate input types, try to launch function
136  launchLogic(cmd, // CommandConstructor pointer, contains target function pointer
137  prm, // ReadCommandFromBuffer Parameters structure reference
138  prm_idx, // Parameters array index
139  tokens_received, // how many tokens were retrieved
140  all_arguments_valid, // type error sentinel
141  match, // function launch sentinel
142  input_type_match_flag, // type error flag array
143  subcommand_matched); // subcommand match flag
144  break; // break command iterator for loop
145  } // end command logic
146  } // end root command for loop
147  if (!match && default_function_ != NULL) // if there was no command match and a default function is configured
148  {
149  _ReadCommandFromBufferErrorOutput(cmd,
150  prm,
151  command_matched,
152  input_type_match_flag,
153  all_arguments_valid,
154  data);
155  (*default_function_)(this); // run the default function
156  }
157  delete[] token_buffer;
158 }

160 void UserInput::GetCommandFromStream(Stream &stream, size_t rx_buffer_size)
161 {
162  if (stream_buffer_allocated == false)
163  {
164  stream_data = new uint8_t[rx_buffer_size]; // an array to store the received data
165  if (stream_data == nullptr) // if there was an error allocating the memory
166  {
167  _ui_out(PSTR(">%s $ERROR: cannot allocate ram for stream_data\n"), _username_);
168  return;
169  }
170  stream_buffer_allocated = true;
171  _term_index_ = 0;
172  }
173  char *rc = (char *)stream_data; // point rc to allocated memory
174  size_t _term_len_ = strlen(_term_);
175  while (stream.available() > 0 && new_stream_data == false)
176  {
177  rc[stream_data_index] = stream.read();
178  if (rc[stream_data_index] == _term_[_term_index_])
179  {
180  stream_data[stream_data_index] = _null_;
181  if (_term_index_ < _term_len_)
182  {
183  _term_index_++;
184  }
185  if (_term_index_ == _term_len_)
186  {
187  new_stream_data = true;
188  }
189  }
190  if (stream_data_index < rx_buffer_size)
191  {
192  stream_data_index++;
193  }
194  }
195  if (new_stream_data == true)
196  {
197  UserInput::ReadCommandFromBuffer(stream_data, stream_data_index);
198  stream_data_index = 0;
199  new_stream_data = false;
200  delete[] stream_data;
201  stream_buffer_allocated = false;
202  }

296 bool UserInput::getToken(char *token_buffer, uint8_t *data, size_t len, size_t *data_index)
297 {
298  bool got_token = false;
299  char incoming = 0; // cast data[data_index] to char and run tests on incoming
300  bool token_flag[2] = {false, false}; // token state machine, point to a token once
301  uint32_t data_length = (uint32_t)len;
302  size_t delim_len = strlen(_delim_);
303  #if defined(__DEBUG_GET_TOKEN__)
304  _ui_out(PSTR("getToken(): "));
305  #endif
306  for (uint16_t i = *data_index; i < data_length; ++i)
307  {
308  incoming = (char)data[*data_index];
309  #if defined(__DEBUG_GET_TOKEN__)
310  char inc_buf[UI_ESCAPED_CHAR_PGM_LEN];
311  _ui_out(PSTR("%s"), (iscntrl(incoming)
312  ? escapeCharactersSoTheyPrint(incoming, *inc_buf)
313  : &incoming));
314  #endif
315  if (iscntrl(incoming)) // remove hanging control characters
316  {
317  token_buffer[*data_index] = _null_;
318  }
319  else if (incoming == _delim_[0]) // replace delimiter
320  {
321  if (delim_len == 1) // incoming is equal to _delim_[0] and the delimiter is one character in length
322  {
323  token_buffer[*data_index] = _null_;
324  token_flag[0] = false;
325  }
326  else
327  {
328  bool delim_char_match = true; // error flag
329  size_t idx = *data_index;
330  char inc;
331  for (size_t j = 1; j < (delim_len + 1); ++j)
332  {
333  if ((j + i) < data_length)
334  {
335  idx++;
336  inc = (char)data[idx];
337  if (inc != _delim_[i])
338  {
339  delim_char_match = false;
340  break; // delimiter char pattern not matched
341  }
342  }
343  else
344  {
345  delim_char_match = false;
346  break; // delimiter char pattern terminated early
347  }
348  }
349  if (delim_char_match == true) // replace delimiter pattern with null
350  {
351  for (size_t j = 1; j < (delim_len + 1); ++j)
352  {
353  (*data_index)++;
354  token_buffer[*data_index] = _null_;
355  }
356  token_flag[0] = false;
357  }
358  }
359  }
360  else if (incoming == *_c_str_delim_) // switch logic for c-string input
361  {
362  token_buffer[*data_index] = _null_; // replace the c-string delimiter
363  if (((uint16_t)(*data_index) + 1U) < data_length) // don't need to do this if we're at the end of user input
364  {
365  bool point_to_beginning_of_c_string = true; // c-string pointer assignment flag
366  (*data_index)++;
367  for (uint16_t j = *data_index; j < data_length; ++j) // this for loop starts at whatever data_index is equal to and has the potential to iterate up to len
368  {
369  incoming = (char)data[*data_index]; // fetch the next incoming char
370  if (incoming == *_c_str_delim_) // if the next incoming char is a '\"'
371  {
372  token_buffer[*data_index] = _null_; // replace the c-string delimiter
373  (*data_index)++; // increment the tokenized string index
374  break;
375  }
376  token_buffer[*data_index] = incoming; // else assign incoming to token_buffer[data_index]
377  if (point_to_beginning_of_c_string == true) // a pointer will be assigned if point_to_beginning_of_c_string == true
378  {
379  data_pointers[data_pointers_index] = &token_buffer[*data_index]; // point to this position in token_buffer
380  data_pointers_index++; // increment pointer index
381  point_to_beginning_of_c_string = false; // only set one pointer per c-string
382  got_token = true;
383  #if defined(__DEBUG_GET_TOKEN__)
384  _ui_out(PSTR("\n>%s $DEBUG: got the c-string token.\n"), _username_);
385  #endif
386  }
387  (*data_index)++; // increment the tokenized string index
388  }
389  }
390  }
391  else // this is a non c-string token
392  {
393  if (incoming == '\\' && (*data_index < data_length))
394  {
395  token_buffer[*data_index] = _null_;
396  (*data_index)++; // increment buffer index
397  incoming = combineControlCharacters((char)data[*data_index]);
398  }
399  token_buffer[*data_index] = incoming; // assign incoming to token_buffer at data_index
400  token_flag[0] = true; // set token available sentinel to true
401  }
402 
403  if (token_flag[0] != token_flag[1]) // if the token state has changed
404  {
405  if (token_flag[0] == true) // if there's a new token
406  {
407  data_pointers[data_pointers_index] = &token_buffer[*data_index]; // assign a pointer the beginning of it
408  data_pointers_index++; // and increment the pointer index
409  }
410  else
411  {
412  #if defined(__DEBUG_GET_TOKEN__)
413  _ui_out(PSTR("\n>%s $DEBUG: getToken got_token == true token state machine.\n"),
414  _username_);
415  #endif
416  got_token = true;
417  break;
418  }
419  token_flag[1] = token_flag[0]; // track the state
420  }
421  if (token_flag[0] == true && (uint16_t)*data_index == (data_length - 1))
422  {
423  #if defined(__DEBUG_GET_TOKEN__)
424  _ui_out(PSTR("\n>%s $DEBUG: getToken() got_token == true end of data.\n"),
425  _username_);
426  #endif
427  got_token = true;
428  }
429 
430  if (*data_index < data_length) // if we are at the end of input data
431  {
432  (*data_index)++; // increment buffer index
433  }
434 
435  if (got_token == true)
436  {
437  #if defined(__DEBUG_GET_TOKEN__)
438  _ui_out(PSTR("\n>%s $DEBUG: getToken() break for loop.\n"), _username_);
439  #endif
440  break;
441  }
442  }
443  return got_token;
444 }

446 bool UserInput::validateUserInput(uint8_t arg_type, size_t data_pointers_index)
447 {
448  uint16_t strlen_data = strlen(data_pointers[data_pointers_index]);
449  bool found_negative_sign = ((char)data_pointers[data_pointers_index][0] == _neg_) ? true : false;
450  if (arg_type < (size_t)UITYPE::CHAR)
451  {
452  // for unsigned integers
453  if (arg_type < (size_t)UITYPE::INT16_T)
454  {
455  if (found_negative_sign == true)
456  {
457  return false;
458  }
459  for (uint16_t j = 0; j < strlen_data; ++j)
460  {
461  if (isdigit(data_pointers[data_pointers_index][j]) == false)
462  {
463  return false;
464  }
465  }
466  }
467  // for integer numbers
468  if (arg_type == (size_t)UITYPE::INT16_T)
469  {
470  if (found_negative_sign == true) // negative
471  {
472  for (uint16_t j = 1; j < (strlen_data - 1); ++j)
473  {
474  if (isdigit(data_pointers[data_pointers_index][j]) == false)
475  {
476  return false;
477  }
478  }
479  }
480  else // positive
481  {
482  for (uint16_t j = 0; j < strlen_data; ++j)
483  {
484  if (isdigit(data_pointers[data_pointers_index][j]) == false)
485  {
486  return false;
487  }
488  }
489  }
490  }
491  // for floating point numbers
492  if (arg_type == (size_t)UITYPE::FLOAT)
493  {
494  uint8_t found_dot = 0;
495  uint8_t num_digits = 0;
496  if (found_negative_sign == true) // negative
497  {
498  uint8_t not_digits = 0;
499  /*
500  we already know there is a '-' at data[i + 1][0] because found_negative_sign is set
501  so start the for loop at an index of one
502  */
503  for (uint16_t j = 1; j < strlen_data; ++j)
504  {
505  if (data_pointers[data_pointers_index][j] == _dot_)
506  {
507  found_dot++;
508  }
509  if (isdigit(data_pointers[data_pointers_index][j]) == true)
510  {
511  num_digits++;
512  }
513  else
514  {
515  not_digits++;
516  }
517  }
518  if (found_dot == 0) // if there is no fraction
519  {
520  found_dot++;
521  }
522  /*
523  if we find one '.' and one '-' or one '-' and no '.' increment num_digits to include the sign, else type match will fail
524  this is to allow negative numbers with no decimal or fraction like "-5" and also input like "-5.0"
525  */
526  if ((not_digits == 2 && found_dot == 1) || (not_digits == 1 && found_dot == 1))
527  {
528  num_digits++; // count the negative sign
529  }
530  }
531  else // positive
532  {
533  for (uint16_t j = 0; j < strlen_data; ++j)
534  {
535  if (data_pointers[data_pointers_index][j] == _dot_)
536  {
537  found_dot++;
538  }
539  if (isdigit(data_pointers[data_pointers_index][j]) == true)
540  {
541  num_digits++;
542  }
543  }
544  }
545  if (found_dot > 1 || (num_digits + found_dot) != strlen_data)
546  {
547  return false;
548  }
549  }
550  }
551  /*
552  For char and c-string input.
553  Types allowed are printable characters, punctuation, control characters \r\n etc, and digits 0-9
554  */
555  else if (arg_type == (size_t)UITYPE::CHAR)
556  {
557  if (strlen_data > 1)
558  {
559  return false;
560  }
561  if (isprint(data_pointers[data_pointers_index][0]) ||
562  ispunct(data_pointers[data_pointers_index][0]) ||
563  iscntrl(data_pointers[data_pointers_index][0]) ||
564  isdigit(data_pointers[data_pointers_index][0]))
565  {
566  }
567  else
568  {
569  return false;
570  }
571  }
572  else if (arg_type == (size_t)UITYPE::C_STRING)
573  {
574  for (uint16_t j = 0; j < strlen_data; ++j)
575  {
576  if (isprint(data_pointers[data_pointers_index][j]) ||
577  ispunct(data_pointers[data_pointers_index][j]) ||
578  iscntrl(data_pointers[data_pointers_index][j]) ||
579  isdigit(data_pointers[data_pointers_index][j]))
580  {
581  }
582  else
583  {
584  return false;
585  }
586  }
587  }
588  else if (arg_type == (size_t)UITYPE::NOTYPE)
589  {
590  // no type validation performed
591  return true;
592  }
593  else // unknown types always return false
594  {
595  return false;
596  }
597  return true;
598 }

651 void UserInput::launchLogic(CommandConstructor *cmd,
652  Parameters &prm,
653  uint8_t &prm_idx,
654  size_t tokens_received,
655  bool &all_arguments_valid,
656  bool &match,
657  bool *input_type_match_flag,
658  bool &subcommand_matched)
659 {
660  // error
661  if (tokens_received > 1 &&
662  prm.sub_commands == 0 &&
663  prm.max_num_args == 0)
664  {
665  #if defined(__DEBUG_LAUNCH_LOGIC__)
666  _ui_out(PSTR(">%s $DEBUG: too many tokens (%d) for command (%s)\n"),
667  _username_, tokens_received, prm.command);
668  #endif
669  return;
670  }
671 
672  // command with no arguments
673  if (subcommand_matched == false &&
674  tokens_received == 1 &&
675  prm.max_num_args == 0)
676  {
677  #if defined(__DEBUG_LAUNCH_LOGIC__)
678  _ui_out(PSTR(">%s $DEBUG: cmd w/no args, subcommand match false "
679  "tokens=0 max_args=0 launchFunction(%s)\n"),
680  _username_, prm.command);
681  #endif
682 
683  match = true; // don't run default callback
684  launchFunction(cmd, prm, prm_idx, tokens_received); // launch the matched command
685  return;
686  }
687 
688  // subcommand with no arguments
689  if (tokens_received == 1 &&
690  _current_search_depth > 1 &&
691  subcommand_matched == true &&
692  prm.max_num_args == 0)
693  {
694  #if defined(__DEBUG_LAUNCH_LOGIC__)
695  _ui_out(PSTR(">%s $DEBUG: subcmd w/no args, subcommand match true "
696  "tokens==0 max_args==0 depth>1 launchFunction(%s)\n"),
697  _username_, prm.command);
698  #endif
699  match = true; // don't run default callback
700  launchFunction(cmd, prm, prm_idx, tokens_received); // launch the matched command
701  return;
702  }
703 
704  // command with arguments, potentially has subcommands but none were entered
705  if (//_current_search_depth > 1 &&
706  subcommand_matched == false &&
707  tokens_received > 1 &&
708  prm.max_num_args > 0)
709  {
710  getArgs(tokens_received, input_type_match_flag, prm, all_arguments_valid);
711  if (rec_num_arg_strings >= prm.num_args &&
712  rec_num_arg_strings <= prm.max_num_args &&
713  all_arguments_valid == true)
714  {
715  #if defined(__DEBUG_LAUNCH_LOGIC__)
716  _ui_out(PSTR(">%s $DEBUG: cmd w/ args, no subcmd match, subcommand match false "
717  "tokens>0 max_args>0 depth>1 launchFunction(%s)\n"),
718  _username_, prm.command);
719  #endif
720  match = true; // don't run default callback
721  launchFunction(cmd, prm, prm_idx, tokens_received); // launch the matched command
722  }
723  // if !match, error
724  return;
725  }
726 
727  // command with arguments (max depth)
728  if (_current_search_depth == (cmd->_tree_depth) &&
729  tokens_received > 1 &&
730  prm.max_num_args > 0 &&
731  prm.sub_commands == 0)
732  {
733  getArgs(tokens_received, input_type_match_flag, prm, all_arguments_valid);
734  // if we received at least min and less than max arguments and they are valid
735  if (rec_num_arg_strings >= prm.num_args &&
736  rec_num_arg_strings <= prm.max_num_args &&
737  all_arguments_valid == true)
738  {
739  #if defined(__DEBUG_LAUNCH_LOGIC__)
740  _ui_out(PSTR(">%s $DEBUG: cmd w/args (max depth) launchFunction(%s)\n"), _username_, prm.command);
741  #endif
742  match = true; // don't run default callback
743  launchFunction(cmd, prm, prm_idx, tokens_received); // launch the matched command
744  }
745  // if !match, error
746  return;
747  }
748 
749  // subcommand search
750  subcommand_matched = false;
751  #if defined(__DEBUG_SUBCOMMAND_SEARCH__)
752  _ui_out(PSTR("search depth (%d)\n"), _current_search_depth);
753  #endif
754  if (_current_search_depth <= (cmd->_tree_depth)) // dig starting at depth 1
755  {
756  // this index starts at one because the parameter array's first element will be the root command
757  for (size_t j = 1; j < (cmd->_param_array_len + 1); ++j) // through the parameter array
758  {
759  memcpy_P(&prm, &(cmd->prm[j]), sizeof(prm));
760  prm_idx = j;
761  if (prm.depth == _current_search_depth)
762  {
763  #if defined(__DEBUG_SUBCOMMAND_SEARCH__)
764  _ui_out(PSTR("match depth cmd=(%s)\ntoken=(%s)\n"),
765  prm.command, data_pointers[data_pointers_index]);
766  #endif
767  if (strcmp(data_pointers[data_pointers_index], prm.command) == 0)
768  {
769  #if defined(__DEBUG_SUBCOMMAND_SEARCH__)
770  _ui_out(PSTR("(%s) subcommand matched, (%d) subcommands, max_num_args (%d)\n"),
771  prm.command, prm.sub_commands, prm.max_num_args);
772  #endif
773  // subcommand matched
774  if (tokens_received > 0)
775  {
776  tokens_received--; // subtract subcommand from tokens received
777  #if defined(__DEBUG_SUBCOMMAND_SEARCH__)
778  _ui_out(PSTR("decrement tokens (%d)\n"), tokens_received);
779  #endif
780  data_pointers_index++;
781  }
782  subcommand_matched = true; // subcommand matched
783  failed_on_subcommand = j; // set error index
784  break; // break out of the loop
785  }
786  }
787  }
788  if (_current_search_depth < (cmd->_tree_depth))
789  {
790  _current_search_depth++;
791  }
792  } // end subcommand search
793  if (subcommand_matched == true) // recursion
794  {
795  #if defined(__DEBUG_SUBCOMMAND_SEARCH__)
796  _ui_out(PSTR("recurse\n"));
797  #endif
798  launchLogic(cmd,
799  prm,
800  prm_idx,
801  tokens_received,
802  all_arguments_valid,
803  match,
804  input_type_match_flag,
805  subcommand_matched);
806  }

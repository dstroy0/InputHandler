157bool UserInput::validateUserInput(UserCallbackFunctionParameters *cmd, uint8_t arg_type, uint16_t data_pointers_index)
158{
159 uint16_t strlen_data = strlen(data_pointers[data_pointers_index]);
160 bool found_negative_sign = ((char)data_pointers[data_pointers_index][0] == *_negative_sign) ? true : false;
161
162 if (arg_type < USER_INPUT_TYPE_CHAR)
163 {
164 // for numbers that are not floating point
165 if (arg_type < USER_INPUT_TYPE_INT16_T)
166 {
167 if (found_negative_sign == true)
168 {
169 return false;
170 }
171 for (uint16_t j = 0; j < strlen_data; ++j)
172 {
173 if (isdigit(data_pointers[data_pointers_index][j]) == false)
174 {
175 return false;
176 }
177 }
178 }
179 // for integer numbers
180 if (arg_type == USER_INPUT_TYPE_INT16_T)
181 {
182 if (found_negative_sign == true)
183 {
184 for (uint16_t j = 1; j < (strlen_data - 1); ++j)
185 {
186 if (isdigit(data_pointers[data_pointers_index][j]) == false)
187 {
188 return false;
189 }
190 }
191 }
192 else
193 {
194 for (uint16_t j = 0; j < strlen_data; ++j)
195 {
196 if (isdigit(data_pointers[data_pointers_index][j]) == false)
197 {
198 return false;
199 }
200 }
201 }
202 }
203 // for floating point numbers
204 if (arg_type == USER_INPUT_TYPE_FLOAT)
205 {
206 uint8_t found_dot = 0;
207 uint8_t num_digits = 0;
208 if (found_negative_sign == true)
209 {
210 uint8_t not_digits = 0;
211 /*
212 we already know there is a '-' at data[i + 1][0] because found_negative_sign is set
213 so start the for loop at an index of one
214 */
215 for (uint16_t j = 1; j < strlen_data; ++j)
216 {
217 if (data_pointers[data_pointers_index][j] == *_dot)
218 {
219 found_dot++;
220 }
221 if (isdigit(data_pointers[data_pointers_index][j]) == true)
222 {
223 num_digits++;
224 }
225 else
226 {
227 not_digits++;
228 }
229 }
230 if (found_dot == 0) // if there is no fraction
231 {
232 found_dot++;
233 }
234 /*
235 if we find one '.' and one '-' or one '-' and no '.' increment num_digits to include the sign, else type match will fail
236 this is to allow negative numbers with no decimal or fraction like "-5" and also input like "-5.0"
237 */
238 if ((not_digits == 2 && found_dot == 1) || (not_digits == 1 && found_dot == 1))
239 {
240 num_digits++; // count the negative sign
241 }
242 }
243 else
244 {
245 for (uint16_t j = 0; j < strlen_data; ++j)
246 {
247 if (data_pointers[data_pointers_index][j] == *_dot)
248 {
249 found_dot++;
250 }
251 if (isdigit(data_pointers[data_pointers_index][j]) == true)
252 {
253 num_digits++;
254 }
255 }
256 }
257 if (found_dot > 1 || (num_digits + found_dot) != strlen_data)
258 {
259 return false;
260 }
261 }
262 }
263 /*
264 For char and c-string input.
265 Types allowed are printable characters, punctuation, control characters \r\n etc, and digits 0-9
266 */
267 else if (arg_type == USER_INPUT_TYPE_CHAR)
268 {
269 if (strlen_data > 1)
270 {
271 return false;
272 }
273 if (isprint(data_pointers[data_pointers_index][0]) ||
274 ispunct(data_pointers[data_pointers_index][0]) ||
275 iscntrl(data_pointers[data_pointers_index][0]) ||
276 isdigit(data_pointers[data_pointers_index][0]))
277 {
278 }
279 else
280 {
281 return false;
282 }
283 }
284 else if (arg_type == USER_INPUT_TYPE_C_STRING)
285 {
286 for (uint16_t j = 0; j < strlen_data; ++j)
287 {
288 if (isprint(data_pointers[data_pointers_index][j]) ||
289 ispunct(data_pointers[data_pointers_index][j]) ||
290 iscntrl(data_pointers[data_pointers_index][j]) ||
291 isdigit(data_pointers[data_pointers_index][j]))
292 {
293 }
294 else
295 {
296 return false;
297 }
298 }
299 }
300 else
301 {
302 return false;
303 }
304
305 return true;
306}

322{

324 {
325 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
326 PSTR(">%s $ERROR: user input exceeds max allowed (UINT16_MAX) input length.\n"), _username_);
327 _output_flag = true;
328 return;
329 }
330 uint16_t data_index = 0; // data iterator
331
332 // should maybe see if there is enough memory to allocate the token buffer
333 token_buffer = new char[len + 1]{'\0'}; // place to chop up the input
334 data_pointers_index = 0; // token buffer pointers
335 rec_num_arg_strings = 0; // number of tokens read from data
336 bool match = false;
337 UserCallbackFunctionParameters *cmd;
338
339 /*
340 this tokenizes an input buffer, it should work with any 8 bit input type that represents char
341 char tokenized_string[] = "A\0Tokenized\0C-string\0"
342 char non_tokenized_string[] = "A Non Tokenized C-string" <-- still has a \0 at the end of the string to terminate it
343 */
344 if (getToken(token_buffer, data, len, &data_index) == true) // if there was a token
345 {
346 bool input_type_match_flag[USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS] = {false};
347 bool all_arguments_valid = true; // error sentinel
348 bool command_matched = false; // error sentinel
349 for (cmd = commands_head_; cmd != NULL; cmd = cmd->next_callback_function_parameters) // iterate through user commands
350 {
351 if (strcmp(data_pointers[0], cmd->command) == 0) // match
352 {
353 command_matched = true;
354 if (cmd->num_args == 0) // command with no arguments
355 {
356 while (getToken(token_buffer, data, len, &data_index) == true && rec_num_arg_strings < USER_INPUT_MAX_NUMBER_OF_COMMAND_ARGUMENTS)
357 {
358 rec_num_arg_strings++;
359 if (rec_num_arg_strings > 0)
360 {
361 break;
362 }
363 }
364#ifdef _DEBUG_USER_INPUT
365 if (EnableDebugOutput == true)
366 {
367 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
368 PSTR(">%s $DEBUG: match zero argument command \"%s\".\n"),
369 _username_,
370 cmd->command);
371 _output_flag = true;
372 }
373#endif
374 match = true; // don't run default callback
375 launchFunction(cmd); // launch the matched command
376 break; // break out of the for loop
377 }
378 else if (cmd->num_args > 0) // cmd->num_args > 0
379 {
380 while (getToken(token_buffer, data, len, &data_index) == true && rec_num_arg_strings < cmd->num_args)
381 {
382
383 input_type_match_flag[rec_num_arg_strings] = validateUserInput(cmd, cmd->_arg_type[rec_num_arg_strings], data_pointers_index - 1); // validate the token
384 if (input_type_match_flag[rec_num_arg_strings] == false) // if the token was not valid input
385 {
386 all_arguments_valid = false; // set the error sentinel to true
387 }
388 rec_num_arg_strings++;
389 }
390 if (rec_num_arg_strings == cmd->num_args && all_arguments_valid == true) // if we received the expected amount of arguments and all of them are valid
391 {
392#ifdef _DEBUG_USER_INPUT
393 if (EnableDebugOutput == true)
394 {
395 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
396 PSTR(">%s $DEBUG: match command \"%s\".\n"),
397 _username_,
398 cmd->command);
399 _output_flag = true;
400 }
401#endif
402 match = true; // don't run default callback
403 launchFunction(cmd); // launch the matched command
404 }
405 }
406 break;
407 }
408 }
409 if (!match && default_handler_ != NULL) // if there was no command match and a default handler is configured
410 {
411 // format a string with useful information
412 char data_char_buffer[len + 1] = {'\0'};
413 for (uint16_t i = 0; i < len; ++i)
414 {
415 data_char_buffer[i] = (char)data[i];
416 }
417 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
418 PSTR(">%s $ERROR: %s\n"),
419 _username_,
420 data_char_buffer);
421 _output_flag = true;
422 if (!command_matched)
423 {
424 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
425 PSTR("Command \"%s\" unknown.\n"),
426 data_pointers[0]);
427 _output_flag = true;
428 }
429 if (command_matched && all_arguments_valid == false)
430 {
431 for (uint8_t i = 0; i < rec_num_arg_strings; ++i)
432 {
433 if (input_type_match_flag[i] == false)
434 {
435
436 char* _type = (char *)pgm_read_dword(&(_input_type_strings[uint8_t(cmd->_arg_type[i])]));
437 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
438 PSTR("\"%s\" argument %u error. Expected a %s; received \"%s\".\n"),
439 cmd->command, i + 1,
440 (char *)_type,
441 data_pointers[i + 1]);
442 }
443 _output_flag = true;
444 }
445 }
446 if (command_matched && rec_num_arg_strings != cmd->num_args)
447 {
448 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
449 PSTR("\"%s\" received %02u arguments; \"%s\" expects %02u arguments.\n"),
450 cmd->command, (rec_num_arg_strings), cmd->command, cmd->num_args);
451 _output_flag = true;
452 }
453 (*default_handler_)(this); // run the default handler
454 }
455 }
456 else
457 {
458 if (len > 0)
459 {
460 (*_string_pos) += snprintf_P(_output_buffer + (*_string_pos), _output_buffer_len,
461 PSTR(">%s $ERROR: No tokens retrieved.\n"), _username_);
462 _output_flag = true;
463 }
464 }
465 delete[] token_buffer;
